## Install

    git clone git@github.com:free4sheng/vimrc.git

    CURR_PATH=$(pwd) &&
    ln -sf $CURR_PATH/vimrc/.vimrc ~/.vimrc &&
    ln -sf $CURR_PATH/vimrc/.vim ~/
